 45def add_user(username: str, password: str) -> tuple[bool, str]:
 46    """
 47    adds user to database\n
 48    **not cached**
 49
 50    Uses 6 queries. (2 max if not valid)
 51    """
 52    if not username.endswith("@case.edu"):
 53        return True, "Username must end with @case.edu"
 54
 55    data_man.connect_to_database()
 56
 57    if data_man.check_for_username(username):
 58        return False, "Username already exists"
 59      
 60    if data_man.check_for_email(username):
 61        return False, "Email already exists"
 62    
 63    
 64    # make user and password
 65    user_id = secrets.token_hex(8)
 66    p = PasswordAttempt(user_id, password)
 67    new_user = User(user_id,username,"",username)
 68    new_password = Password(p.hash,p.salt,user_id)
 69
 70    data_man.add_object(new_user)
 71    data_man.add_object(new_password)
 72
 73    # Make code

 75    new_code = Codes(user_id,code)
 76    data_man.add_object(new_code)

 97def make_all_listing_data(listings: list[Listing]) -> list[dict]:
 98    """makes data for all listings
 99
100    Args:
101        listings (list[Listing]): all listings we want to get data for
102
103    Returns:
104        list[dict]: all data for the listing
105
106    Uses 3 queries. (assuming no data is cached)
107    """
108
109    # get all data required for this (3 queries)
110    users = data_man.get_all_from(User())
111    comments = data_man.get_all_from(Comments())
112    ar = data_man.get_all_from(AverageRating())
113
114    # compute and compress data
115
116    comments_by_listing:dict[str, list[Comments]] = {}
117    for com in comments:
118        if com.ListingID in comments_by_listing:
119            comments_by_listing[com.ListingID].append(com)
120            continue
121        comments_by_listing[com.ListingID] = [com]
122
123    user_by_userid:dict[str, User] = {}
124    for usr in users:
125        if usr.UserID in user_by_userid:
126            continue
127        user_by_userid[usr.UserID] = usr
128
129    average_by_listing:dict[str, AverageRating] = {}
130    for average in ar:
131        if average.ListingID in average_by_listing:
132            continue
133        average_by_listing[average.ListingID] = average
134
135
136    users_by_comments:dict[str, list[User]] = {}
137    for com in comments:
138        if com.UserID in users_by_comments:
139            users_by_comments[com.UserID].append(user_by_userid[com.UserID])
140            continue
141        try:
142            users_by_comments[com.UserID] = [user_by_userid[com.UserID]]
143        except:
144            continue
145
146
147    listing_data = []
148    for listing in listings:
149        # check for listing in chache
150        try:
151            if f"listing_{listing.ListingID}" in cache_man.all_refrences:
152                cache_data = cache_man.get_cache(f"listing_{listing.ListingID}")
153                if cache_data.cache_max_age > datetime.now():
154                    # use cached data
155                    raw_data:dict = cache_data.cache_data
156                    meta_listing:dict = raw_data["meta_listing"]
157                    comments_with_users:list = raw_data["comments"]
158                    listing_data.append({
159                        "listing": meta_listing,
160                        "comments": comments_with_users
161                    })
162                    continue
163        except:
164            pass
165
166        # meta listing creation
167        meta_listing = listing.as_dict()
168        try:
169            comments = comments_by_listing[listing.ListingID]
170        except:
171            comments = []
172        ar = average_by_listing[listing.ListingID]
173        
174        # new variables being sent to website
175        meta_listing["avg_rating"] = round(ar.AverageRating, 2)
176        meta_listing["review_count"] = ar.NumberOfRatings
177        created_str = ""
178        if listing.CreatedAt:
179            try:
180                utc_dt = datetime.fromisoformat(listing.CreatedAt.replace("Z", "")).replace(tzinfo=timezone.utc)
181                eastern_dt = utc_dt.astimezone(ZoneInfo("America/New_York"))
182                created_str = eastern_dt.strftime("%m/%d/%Y, %I:%M %p")
183            except:
184                created_str = listing.CreatedAt
185        meta_listing["CreatedAt"] = created_str
186
187        # comments
188        comments_with_users = []
189        for c in comments:
190            try:
191                user = data_man.get_user_from_comments(c)
192                # Convert comment timestamp to Eastern Time
193                if c.CreatedAt:
194                    utc_dt = datetime.fromisoformat(c.CreatedAt.replace("Z", "")).replace(tzinfo=timezone.utc)
195                    eastern_dt = utc_dt.astimezone(ZoneInfo("America/New_York"))
196                    created_str = eastern_dt.strftime("%m/%d/%Y, %I:%M %p")
197                else:
198                    created_str = ""
199                comments_with_users.append({
200                    "Content": c.Content,
201                    "Username": user.Username,
202                    "CreatedAt": created_str,
203                    "Tags": c.Tags if c.Tags else []
204                })
205            except TypeError as e:
206                print(f"user failed: {e}")
207                comments_with_users.append({
208                    "Content": c.Content,
209                    "Username": "Unknown",
210                    "CreatedAt": "",
211                    "Tags": c.Tags if c.Tags else []
212                })
213
214        # pack into cached data
215        data = {"meta_listing":meta_listing,"comments":comments_with_users}
216        cache_man.add_to_cache(data,f"listing_{listing.ListingID}")
217        listing_data.append({
218            "listing": meta_listing,
219            "comments": comments_with_users
220        })
221    return listing_data
